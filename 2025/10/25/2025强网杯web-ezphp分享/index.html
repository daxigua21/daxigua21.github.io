function generateRandomString($length = 8) {
    $characters = 'abcdefghijklmnopqrstuvwxyz';
    $randomString = '';
    for ($i = 0; $i < $length; $i++) {
        $r = rand(0, strlen($characters) - 1);
        $randomString .= $characters[$r];
    }
    return $randomString;
}

иҝӣиЎҢд»Јз Ғе®Ўи®ЎпјҢ__constructжҳҜеҜ№дёҠдј зҡ„ж–Үд»¶иҝӣиЎҢеӨ„зҗҶпјҢreadflagжҳҜиҝӣиЎҢincludeж–Үд»¶зҡ„пјҢз„¶еҗҺеҲ©з”Ё__destructиҝҷйҮҢзҡ„ifиҜӯеҸҘпјҢжү§иЎҢжҳҜ

new $func()е’Ң$func()ж“ҚдҪңпјҢжүҖд»ҘеӨ§иҮҙжҖқи·Ҝе°ұжҳҜдёҠдј ж–Үд»¶пјҢж–Үд»¶еҢ…еҗ«

и§ҰеҸ‘__constructеҸҜд»Ҙз”Ё__destructдёӯзҡ„new $func()е®һдҫӢеҢ–и§ҰеҸ‘пјҢ__constructйҮҢйқўзҡ„еҢҝеҗҚзұ»дёӯзҡ„__constructд№ҹеҸҜд»ҘзӣҙжҺҘи§ҰеҸ‘

жҖҺд№Ҳи§ҰеҸ‘readflagе‘ўпјҢиҝҷйҮҢеңЁ$this->key == 'func'зҡ„жғ…еҶөдёӢзӣҙжҺҘи®©$func=readflagзҡ„иҜқжҳҜиӮҜе®ҡдёҚиЎҢзҡ„пјҢеӣ дёәдёҚеңЁдёҖдёӘдҪңз”ЁеҹҹйҮҢ

жңүиҝҷд№ҲдёҖдёӘпјҢarray(зұ»еҗҚпјҢ'еҮҪж•°еҗҚ')()еҸҜд»ҘзӣҙжҺҘи°ғз”Ёзұ»дёӯзҡ„еҮҪж•°,зұ»дјјдәҺcall_user_func(array(зұ»еҗҚ,'еҮҪж•°еҗҚ')пјҢеҸӮж•°)

readflagеҮҪж•°пјҢеӨ–еұӮзҡ„еҸҜд»Ҙз”Ёarray()жү§иЎҢпјҢеҶ…еұӮзҡ„readflagеңЁи°ғз”Ёе®ҢеӨ–еұӮзҡ„readflagд№ӢеҗҺеҸҜд»ҘзӣҙжҺҘеңЁе…ЁеұҖдёӯи°ғз”Ё

еӣ дёәеӨ–еұӮзҡ„readflagжү§иЎҢе®ҢеҗҺдјҡе°ҶеҮҪж•°еҶ…зҡ„еҮҪж•°еЈ°жҳҺдёәе…ЁеұҖзҡ„пјҢе°ұеғҸдёҠеӣҫдёӯгҖӮ

жүҖд»ҘжҲ‘д»¬е…Ҳжү§иЎҢеӨ–еұӮзҡ„readflagпјҢеЈ°жҳҺеҗҺзӣҙжҺҘи°ғз”ЁеҶ…еұӮзҡ„readflagпјҢжҺҘдёӢжқҘиҝӣиЎҢеәҸеҲ—еҢ–ж“ҚдҪңпјҢеӣ дёәеӨ–еұӮеңЁеҢҝеҗҚеҮҪж•°дёӯпјҢжҲ‘д»¬еҸҜд»ҘдҪҝз”ЁжҢҮй’Ҳеј•з”Ёзҡ„ж–№ејҸжү§иЎҢ

жү§иЎҢ$aзҡ„__destruct->$bзҡ„__construct->$bзҡ„__destructпјҢеҰӮжһңжІЎжңүиҝҷйҮҢзҡ„жҢҮй’Ҳеј•з”Ё$b->f=[&$b->aaa,'readflag'];пјҢ$b->aaa=nullпјҢ$b->f=nullпјҢз„¶еҗҺжү§иЎҢ$b->fзҡ„readflagе°ұдјҡжҠҘй”ҷпјҢеӣ дёә$b->fжҳҜnullпјҢдҪҶжҳҜеҰӮжһңдҪҝз”ЁдәҶжҢҮй’Ҳеј•з”ЁпјҢиҷҪз„¶дёҖејҖе§ӢжҳҜnullпјҢдҪҶжҳҜеңЁеҗҺйқўжү§иЎҢдәҶ$bзҡ„__constructпјҢ$bзҡ„aaaе°ұдёҚжҳҜnullпјҢ$b->f[0]еӣ дёәжҳҜеҗҢдёҖдёӘеј•з”ЁпјҢжүҖд»ҘдјҡдёҖиө·ж”№еҸҳ,жү§иЎҢ$func();пјҢ$b->f[0]е°ұдёҚдјҡеӣ дёәжҳҜnullиҖҢжҠҘй”ҷдәҶ
